-- 1. суммарное количество уникальных предприятий, цехов и агрегатов
SELECT
    COUNT(DISTINCT p.NAME) AS Total_Enterprises,
    COUNT(DISTINCT s.NAME) AS Total_Shops,
    COUNT(m.NAME) AS Total_Machines
FROM TPLANTS p
LEFT JOIN TSHOPS s ON p.PLANT_ID = s.PLANT_ID
LEFT JOIN TMACHINES m ON s.SHOP_ID = m.SHOP_ID;


-- 2. список предприятий и количество агрегатов на каждом из них
SELECT
    p.NAME AS Plant_Name,
    COUNT(m.NAME) AS Total_Machines
FROM TPLANTS p
JOIN TSHOPS s ON p.PLANT_ID = s.PLANT_ID
JOIN TMACHINES m ON s.SHOP_ID = m.SHOP_ID
GROUP BY p.NAME
ORDER BY Total_Machines DESC;


-- 3. общее количество замеров вибрации в базе и даты самого старого и самого свежего измерения
SELECT
    COUNT(MEASUREMENT_ID) AS Total_Measurements,
    MIN(MEASURE_TIME) AS Oldest_Record,
    MAX(MEASURE_TIME) AS Newest_Record
FROM TMEASUREMENTS;


-- 4. общее количество прикрепленных файлов осциллограмм (.afdata)
SELECT
    COUNT(*) AS Total_Wave_Files
FROM TMEASUREMENT_FILES_INFO;


-- 5. Показывает количество фактически снятых числовых значений вибрации за конкретный год
SELECT
    COUNT(v.PARAM_VALUE_ID) AS Total_Numeric_Values_2024
FROM TMEASUREMENTS m
JOIN TPARAM_VALUES v ON m.MEASUREMENT_ID = v.MEASUREMENT_ID
WHERE YEAR(m.MEASURE_TIME) = 2024;
